CREATE TABLE account
(
	account_id           VARCHAR2(50) NOT NULL ,
	user_id              VARCHAR2(10) NOT NULL ,
	account_type         VARCHAR2(20) NOT NULL ,
	bank                 VARCHAR2(20) NOT NULL ,
	account_name         VARCHAR2(20) NOT NULL 
);

CREATE TABLE manage
(
	account_id           VARCHAR2(50) NOT NULL ,
	user_id              VARCHAR2(10) NOT NULL ,
	sm_join              DATE NOT NULL ,
	sm_nap               NUMBER(15) NOT NULL ,
	sm_napgi             DATE NOT NULL ,
	sm_rate              NUMBER(10) NOT NULL 
);

CREATE TABLE tran
(
	account_id           VARCHAR2(50) NOT NULL ,
	user_id              VARCHAR2(10) NOT NULL ,
	t_date               DATE NOT NULL ,
	withdraw             NUMBER(15) NULL ,
	deposit              NUMBER(15) NULL ,
	t_contents           VARCHAR2(50) NULL ,
	balance              NUMBER(15) NOT NULL 
);

CREATE TABLE user1
(
	user_id              VARCHAR2(10) NOT NULL ,
	user_pw              VARCHAR2(10) NOT NULL ,
	user_name            VARCHAR2(10) NOT NULL ,
	user_birth           DATE NOT NULL ,
	user_sex             CHAR(18) NOT NULL ,
	limit_money          NUMBER(15) NULL ,
	goal_long            DATE NULL ,
	goal_money           NUMBER(15) NULL 
);
